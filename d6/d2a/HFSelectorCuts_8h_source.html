26 // accounts for the offset so that pt bin array can be used to also configure a histogram axis
27 template <typename T1, typename T2>
28 int findBin(o2::framework::Configurable<std::vector<T1>> const& bins, T2 value)
29 {
30  if (value < bins->front()) {
31  return -1;
32  }
33  if (value >= bins->back()) {
34  return -1;
35  }
36  int step;
37  int bin = 0;
38  int count = bins->size();
39 
40  while (count > 0) {
41  step = count / 2;
42  bin += step;
43  if (bins->operator[](bin) <= value) {
44  count -= step + 1;
45  } else {
46  bin -= step;
47  count = step;
48  }
49  }
50  if (bin == bins->size()) {
51  bin = 0;
52  }
53  return bin - 1;
54 }
